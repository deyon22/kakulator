function input(value) { if (current === "0") current = ""; current += value; display.innerText = current; } function clearAll() { current = ""; display.innerText = "0"; } function backspace() { current = current.slice(0, -1); display.innerText = current || "0"; } function calculate() { try { current = eval(current).toString(); display.innerText = current; } catch { display.innerText = "Error"; current = ""; } }
